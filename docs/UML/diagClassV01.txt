enum TYPE {
  EMPTY 0
  FLOOR 1
  GROUND 2
}

enum Action {
  WALK_FRONT 0
  WALK_BACK 1
  STAY 2
  DIG_BELOW 3
  DIG_FRONT 4
  DIG_BACK 5
}

class QState{
 -Type left
 -Type center
 -Type right
 -Type bottom 
 +boolean is(Type)
}

QState --o QLink
class QLink {
 -QState from
 -QState to
 -int[Action.number] actionRewards
}
note right
The index of a case of
<i>actionRewards</i> is the 
corresponding action index.
end note

QLink  --* QGraph
class QGraph {
 -LinkedList<LinkedList<QLink>> stateNodesList
}
note right
The index of a QLink of
<i>stateNodesList</i> is the 
corresponding state index.
Each QLink at the index i
of <i>stateNodesList</i> has
as <i>from</i> the state i.
end note

@enduml
